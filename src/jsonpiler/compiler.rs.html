12  /// Builds the assembly code from the parsed JSON.
13  /// This function is the main entry point for the compilation.
14  /// It takes the parsed JSON, sets up the initial function table,
15  /// evaluates the JSON, and writes the resulting assembly code to a file.
16  /// # Arguments
17  /// * `source` - The JSON string.
18  /// * `json_file` - The name of the original JSON file.
19  /// * `out_file` - The name of the file to write the assembly code to.
20  /// # Returns
21  /// * `Ok(())`
22  /// * `Err(Box<dyn Error>)` - If an error occurred during the compilation.
23  /// # Errors
24  /// * `Box<dyn Error>` - If an error occurred during the compilation.
25  #[inline]
26  pub fn build(&mut self, source: String, json_file: &str, out_file: &str) -> ErrOR<()> {
27    let json = self.parse(source)?;
28    self.include_flag = HashSet::new();
29    self.sect = Section::default();
30    self.symbol_seeds = HashMap::new();
31    self.builtin = HashMap::new();
32    self.vars = vec![HashMap::new()];
33    self.all_register();
34    let mut info = FuncInfo::default();
35    let result = self.eval(&json, &mut info)?;
36    let mut writer = io::BufWriter::new(File::create(out_file)?);
37    writer.write_all(format!(".file \"{json_file}\"\n.intel_syntax noprefix\n").as_bytes())?;
38    writer.write_all(include_bytes!("asm/sect/data.s"))?;
39    for data in &mut self.sect.data {
40      writer.write_all(data.as_bytes())?;
41    }
42    writer.write_all(include_bytes!("asm/sect/bss.s"))?;
43    for bss in &mut self.sect.bss {
44      writer.write_all(bss.as_bytes())?;
45    }
46    writer.write_all(include_bytes!("asm/sect/start.s"))?;
47    writer.write_all(
48      format!(include_str!("asm/common/prologue.s"), size = info.calc_alloc(8)?).as_bytes(),
49    )?;
50    writer.write_all(include_bytes!("asm/sect/startup.s"))?;
51    for body in &mut info.body {
52      writer.write_all(body.as_bytes())?;
53    }
54    if let Json::LInt(int) = result.value {
55      writer.write_all(format!("  mov rcx, {int}\n").as_bytes())
56    } else if let Json::VInt(var) = &result.value {
57      writer.write_all(format!("  mov rcx, {var}\n").as_bytes())
58    } else {
59      writer.write_all(b"  xor ecx, ecx\n")
60    }?;
61    writer.write_all(b"  call [qword ptr __imp_ExitProcess[rip]]\n.seh_endproc\n")?;
62    writer.write_all(include_bytes!("asm/sect/handler.s"))?;
63    for text in &mut self.sect.text {
64      writer.write_all(text.as_bytes())?;
65    }
66    writer.flush()?;
67    Ok(())
68  }
69  /// Evaluates a JSON object.
70  pub(crate) fn eval(&mut self, json: &JsonWithPos, info: &mut FuncInfo) -> JResult {
71    const ERR: &str = "Unreachable (eval)";
72    let Json::LArray(list) = &json.value else {
73      let Json::LObject(object) = &json.value else { return Ok(json.clone()) };
74      let mut evaluated = JObject::default();
75      for kv in object.iter() {
76        evaluated.insert(kv.0.clone(), self.eval(&kv.1, info)?);
77      }
78      return Ok(JsonWithPos { value: Json::LObject(evaluated), pos: json.pos.clone() });
79    };
80    self.validate_args("function call", true, 1, list.len(), &json.pos)?;
81    let first_elem = list.first().ok_or(ERR)?;
82    let first = &self.eval(first_elem, info)?;
83    if let Json::LString(cmd) = &first.value {
84      if let Ok(Json::Function(af)) = self.get_var(cmd, &first.pos) {
85        info.body.push(format!("  call {}\n", af.name));
86        if let Json::VInt(_) | Json::LInt(_) = *af.ret {
87          let name = self.get_name("BSS", "8")?;
88          info.body.push(format!("  mov {name}, rax\n"));
89          Ok(JsonWithPos { value: Json::VInt(name), pos: first.pos.clone() })
90        } else {
91          Ok(JsonWithPos::default())
92        }
93      } else if self.builtin.contains_key(cmd.as_str()) {
94        let builtin = self.builtin.get_mut(cmd.as_str()).ok_or(ERR)?;
95        let scoped = builtin.scoped;
96        let func = builtin.func;
97        if scoped {
98          self.vars.push(HashMap::new());
99        }
100        let rest = list.get(1..).unwrap_or(&[]);
101        let args = if builtin.skip_eval { rest } else { &self.eval_args(rest, info)? };
102        let result = func(self, first, args, info)?;
103        if scoped {
104          self.vars.pop();
105        }
106        Ok(JsonWithPos { value: result, pos: first.pos.clone() })
107      } else {
108        Err(self.fmt_err(&format!("The `{cmd}` function is undefined."), &first.pos).into())
109      }
110    } else if let Json::Function(af) = &first.value {
111      info.body.push(format!("  call {}", af.name));
112      if let Json::VInt(_) | Json::LInt(_) = *af.ret {
113        let name = self.get_name("BSS", "8")?;
114        info.body.push(format!("  mov {name}, rax\n"));
115        Ok(JsonWithPos { value: Json::VInt(name), pos: first.pos.clone() })
116      } else {
117        Ok(JsonWithPos::default())
118      }
119    } else {
120      self.typ_err(1, "function call", "LString` or `Function", first_elem)?;
121      Ok(JsonWithPos::default())
122    }
123  }
124  /// Evaluate arguments.
125  fn eval_args(&mut self, args: &Args, info: &mut FuncInfo) -> ErrOR<Vec<JsonWithPos>> {
126    let mut result = vec![];
127    for arg in args {
128      result.push(self.eval(arg, info)?);
129    }
130    Ok(result)
131  }
132  /// Generates a unique name for internal use.
133  pub(crate) fn get_name(&mut self, name: &str, value: &str) -> Result<String, String> {
134    let seed = self
135      .symbol_seeds
136      .get(name)
137      .map_or(Ok(0), |current| current.checked_add(1).ok_or("Seed Overflow"))?;
138    self.symbol_seeds.insert(name.to_owned(), seed);
139    let l_name = format!(".L{name}{seed:x}");
140    let name_fmt = format!("qword ptr {l_name}[rip]");
141    match name {
142      "BSS" => self.sect.bss.push(format!("  .lcomm {l_name}, {value}\n")),
143      "STR" => {
144        if let Some(str_seed) = self.str_cache.get(value) {
145          return Ok(format!("qword ptr .LSTR{str_seed:x}[rip]"));
146        }
147        self.str_cache.insert(value.to_owned(), self.symbol_seeds.len());
148        self.sect.data.push(format!("  {l_name}: .string \"{value}\"\n"));
149      }
150      "INT" => self.sect.data.push(format!("  {l_name}: .quad {value}\n")),
151      "FNC" => return Ok(l_name),
152      _ => return Err(format!("Internal Error: Unrecognized name: {name}")),
153    }
154    Ok(name_fmt)
155  }
156  /// Gets variable.
157  pub(crate) fn get_var(&self, var_name: &str, pos: &Position) -> ErrOR<Json> {
158    for scope in self.vars.iter().rev() {
159      if let Some(val) = scope.get(var_name) {
160        return Ok(val.clone());
161      }
162    }
163    err!(self, pos, "Undefined variables: `{var_name}`")
164  }
165  /// Registers a function in the function table.
166  pub(crate) fn register(&mut self, name: &str, flag: (bool, bool), j_func: JFunc) {
167    self.builtin.insert(name.into(), Builtin { skip_eval: flag.0, scoped: flag.1, func: j_func });
168  }
169  /// Converts `JValue` (`StringVar` or `String`) to `StringVar`, otherwise return `Err`
170  pub(crate) fn string2var(
171    &mut self, json: &JsonWithPos, ordinal: usize, func_name: &str,
172  ) -> ErrOR<String> {
173    if let Json::LString(l_str) = &json.value {
174      let name = self.get_name("STR", l_str)?;
175      Ok(name)
176    } else if let Json::VString(v_str) = &json.value {
177      Ok(v_str.clone())
178    } else {
179      self.typ_err(ordinal, func_name, "String", json)?;
180      Ok(String::new())
181    }
182  }
183  /// Generates a type error.
184  pub(crate) fn typ_err(
185    &self, ordinal: usize, name: &str, expected: &str, json: &JsonWithPos,
186  ) -> FResult {
187    let suffix = match ordinal % 100 {
188      11..=13 => "th",
189      _ => match ordinal % 10 {
190        1 => "st",
191        2 => "nd",
192        3 => "rd",
193        _ => "th",
194      },
195    };
196    let typ = json.value.type_name();
197    err!(
198      self,
199      &json.pos,
200      "The {ordinal}{suffix} argument to `{name}` must be of a type `{expected}`, \
201      but a value of type `{typ}` was provided."
202    )
203  }
204  /// Generate an error.
205  pub(crate) fn validate_args(
206    &self, name: &str, at_least: bool, expected: usize, supplied: usize, pos: &Position,
207  ) -> ErrOR<()> {
208    let fmt_require = |text: &str| -> ErrOR<()> {
209      let (plural, be) = if expected == 1 { ("", "is") } else { ("s", "are") };
210      err!(
211        self,
212        pos,
213        "`{name}` requires {text} {expected} argument{plural}, \
214        but {supplied} argument{plural} {be} supplied.",
215      )
216    };
217    if at_least {
218      if supplied >= expected { Ok(()) } else { fmt_require("at least") }
219    } else if expected == supplied {
220      Ok(())
221    } else {
222      fmt_require("exactly")
223    }
224  }
