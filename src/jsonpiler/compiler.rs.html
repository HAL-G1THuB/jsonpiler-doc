15impl Jsonpiler {
16  /// Builds the assembly code from the parsed JSON.
17  /// This function is the main entry point for the compilation.
18  /// It takes the parsed JSON, sets up the initial function table,
19  /// evaluates the JSON, and writes the resulting assembly code to a file.
20  /// # Arguments
21  /// * `source` - The JSON string.
22  /// * `json_file` - The name of the original JSON file.
23  /// * `out_file` - The name of the file to write the assembly code to.
24  /// # Returns
25  /// * `Ok(())`
26  /// * `Err(Box<dyn Error>)` - If an error occurred during the compilation.
27  /// # Errors
28  /// * `Box<dyn Error>` - If an error occurred during the compilation.
29  #[inline]
30  pub fn build(&mut self, source: &str, out_file: &str) -> ErrOR<()> {
31    let json = self.parse(source)?;
32    self.include_flag = HashSet::new();
33    self.text = vec![];
34    self.bss = vec![];
35    self.data = vec![];
36    self.global_seed = 0;
37    self.builtin = HashMap::new();
38    self.vars = vec![HashMap::new()];
39    self.all_register();
40    let mut info = FuncInfo::default();
41    let result = self.eval(json.value, &mut info)?;
42    let mut writer = io::BufWriter::new(File::create(out_file)?);
43    writer.write_all(mn(".intel_syntax", &["noprefix"]).as_bytes())?;
44    writer.write_all(include_bytes!("asm/once/data.s"))?;
45    for data in &mut self.data {
46      writer.write_all(data.as_bytes())?;
47    }
48    writer.write_all(include_bytes!("asm/once/bss.s"))?;
49    for bss in &mut self.bss {
50      writer.write_all(bss.as_bytes())?;
51    }
52    writer.write_all(include_bytes!("asm/once/main.s"))?;
53    write!(
54      writer,
55      include_str!("asm/common/prologue.s"),
56      size = format!("0x{:x}", info.calc_alloc(8)?)
57    )?;
58    writer.write_all(include_bytes!("asm/once/startup.s"))?;
59    for body in &mut info.body {
60      writer.write_all(body.as_bytes())?;
61    }
62    if let Json::Int(int) = result {
63      match int {
64        Bind::Lit(l_int) => writer.write_all(mn("mov", &["rcx", &l_int.to_string()]).as_bytes()),
65        Bind::Var(name) => {
66          writer.write_all(mn("mov", &["rcx", &format!("qword{name}")]).as_bytes())
67        }
68      }
69    } else {
70      writer.write_all(mn("xor", &["ecx", "ecx"]).as_bytes())
71    }?;
72    writer.write_all(imp_call("ExitProcess").as_bytes())?;
73    writer.write_all(mn(".seh_endproc", &[]).as_bytes())?;
74    for text in &mut self.text {
75      writer.write_all(text.as_bytes())?;
76    }
77    writer.write_all(include_bytes!("asm/once/handler.s"))?;
78    writer.flush()?;
79    Ok(())
80  }
81  /// Evaluate JSON representation.
82  pub(crate) fn eval(&mut self, mut json: Json, info: &mut FuncInfo) -> ErrOR<Json> {
83    match &mut json {
84      Json::Array(Bind::Lit(list)) => {
85        Ok(Json::Array(Bind::Lit(self.eval_args(mem::take(list), info)?)))
86      }
87      Json::Object(Bind::Lit(object)) => {
88        let mut result = Json::Null;
89        for (key, val) in object.iter_mut() {
90          let func_name = key.clone();
91          let func_pos = val.pos.clone();
92          let func_expr = JsonWithPos {
93            value: Json::String(Bind::Lit(func_name.clone())),
94            pos: func_pos.clone(),
95          };
96          if let Some(builtin) = self.builtin.get_mut(&func_name) {
97            let scoped = builtin.scoped;
98            let func = builtin.func;
99            let mut tmp = FuncInfo::default();
100            if scoped {
101              self.vars.push(HashMap::new());
102              scope_begin(&mut tmp, info)?;
103            }
104            let args = match &mut val.value {
105              Json::Array(Bind::Lit(arr)) => {
106                let raw_args = mem::take(arr);
107                if self.builtin.get(&func_name).is_some_and(|built| built.skip_eval) {
108                  raw_args
109                } else {
110                  self.eval_args(raw_args, info)?
111                }
112              }
113              _ => self.eval_args(vec![mem::take(val)], info)?,
114            };
115            result = func(self, &func_expr, args, info)?;
116            if let Some((addr, size)) = result.tmp() {
117              info.free(addr, size)?;
118            }
119            if scoped {
120              scope_end(&mut tmp, info)?;
121              self.vars.pop();
122            }
123          } else if let Ok(Json::Function(func)) = self.get_var(&func_name, &func_pos) {
124            const MOV: &str = "  mov ";
125            info.body.push(format!("  call .L{:x}\n", func.name));
126            result = if let Json::Int(_) = *func.ret {
127              let name = info.get_local(8)?;
128              info.body.push(format!("{MOV}qword{name}, rax\n"));
129              Json::Int(Bind::Var(Name { var: Tmp, seed: name.seed }))
130            } else {
131              Json::Null
132            }
133          } else {
134            return err!(self, func_pos, "The `{func_name}` function is undefined.");
135          }
136        }
137        Ok(result)
138      }
139      _ => Ok(json),
140    }
141  }
142  /// Evaluate arguments.
143  fn eval_args(&mut self, mut args: Args, info: &mut FuncInfo) -> ErrOR<Vec<JsonWithPos>> {
144    for arg in &mut args {
145      let with_pos = mem::take(arg);
146      arg.pos = with_pos.pos.clone();
147      arg.value = self.eval(with_pos.value, info)?;
148    }
149    Ok(args)
150  }
151  /// Generates a unique name for internal use.
152  pub(crate) fn get_global(&mut self, name: &GlobalKind, value: &str) -> ErrOR<Name> {
153    let seed = self.global_seed;
154    match name {
155      GlobalKind::Bss => self.bss.push(format!("  .lcomm .L{seed:x}, {value}\n")),
156      GlobalKind::Str => {
157        if let Some(str_seed) = self.str_cache.get(value) {
158          return Ok(Name { var: Global, seed: *str_seed });
159        }
160        self.str_cache.insert(value.to_owned(), seed);
161        self.data.push(format!("  .L{seed:x}: .string \"{value}\"\n"));
162      }
163      GlobalKind::Int => self.data.push(format!("  .L{seed:x}: .quad {value}\n")),
164      GlobalKind::Fnc => (),
165    }
166    self.global_seed = add(self.global_seed, 1)?;
167    Ok(Name { var: Global, seed })
168  }
169  /// Gets variable.
170  pub(crate) fn get_var(&self, var_name: &str, pos: &Position) -> ErrOR<Json> {
171    for scope in self.vars.iter().rev() {
172      if let Some(val) = scope.get(var_name) {
173        return Ok(val.clone());
174      }
175    }
176    err!(self, pos, "Undefined variables: `{var_name}`")
177  }
178  /// Registers a function in the function table.
179  pub(crate) fn register(&mut self, name: &str, flag: (bool, bool), func: JFunc) {
180    self.builtin.insert(name.into(), Builtin { skip_eval: flag.0, scoped: flag.1, func });
181  }
