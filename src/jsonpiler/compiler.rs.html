13impl Jsonpiler {
14  /// Builds the assembly code from the parsed JSON.
15  /// This function is the main entry point for the compilation.
16  /// It takes the parsed JSON, sets up the initial function table,
17  /// evaluates the JSON, and writes the resulting assembly code to a file.
18  /// # Arguments
19  /// * `source` - The JSON string.
20  /// * `json_file` - The name of the original JSON file.
21  /// * `out_file` - The name of the file to write the assembly code to.
22  /// # Returns
23  /// * `Ok(())`
24  /// * `Err(Box<dyn Error>)` - If an error occurred during the compilation.
25  /// # Errors
26  /// * `Box<dyn Error>` - If an error occurred during the compilation.
27  #[inline]
28  pub fn build(&mut self, source: String, json_file: &str, out_file: &str) -> ErrOR<()> {
29    let json = self.parse(source)?;
30    self.include_flag = HashSet::new();
31    self.sect = Section::default();
32    self.symbol_seeds = HashMap::new();
33    self.builtin = HashMap::new();
34    self.vars = vec![HashMap::new()];
35    self.all_register();
36    let mut info = FuncInfo::default();
37    let result = self.eval(json, &mut info)?;
38    let mut writer = io::BufWriter::new(File::create(out_file)?);
39    writeln!(writer, ".file \"{json_file}\"\n.intel_syntax noprefix")?;
40    writer.write_all(include_bytes!("asm/once/data.s"))?;
41    for data in &mut self.sect.data {
42      writer.write_all(data.as_bytes())?;
43    }
44    writer.write_all(include_bytes!("asm/once/bss.s"))?;
45    for bss in &mut self.sect.bss {
46      writer.write_all(bss.as_bytes())?;
47    }
48    writer.write_all(include_bytes!("asm/once/main.s"))?;
49    write!(writer, include_str!("asm/common/prologue.s"), size = info.calc_alloc(8)?)?;
50    writer.write_all(include_bytes!("asm/once/startup.s"))?;
51    for body in &mut info.body {
52      writer.write_all(body.as_bytes())?;
53    }
54    if let Json::Int(int) = result.value {
55      match int {
56        Bind::Lit(l_int) => writeln!(writer, "  mov rcx, {l_int}"),
57        Bind::Var(var) => writeln!(writer, "  mov rcx, {var}"),
58        Bind::Local(local) | Bind::Tmp(local) => {
59          writeln!(writer, "  mov rcx, {}", fmt_local("qword", local))
60        }
61      }
62    } else {
63      writeln!(writer, "  xor ecx, ecx")
64    }?;
65    writer.write_all(b"  call [qword ptr __imp_ExitProcess[rip]]\n.seh_endproc\n")?;
66    for text in &mut self.sect.text {
67      writer.write_all(text.as_bytes())?;
68    }
69    writer.write_all(include_bytes!("asm/once/handler.s"))?;
70    writer.flush()?;
71    Ok(())
72  }
73  /// Evaluates a JSON object.
74  pub(crate) fn eval(&mut self, mut json: JsonWithPos, info: &mut FuncInfo) -> JResult {
75    const ERR: &str = "Unreachable (eval)";
76    let Json::Array(Bind::Lit(list)) = &mut json.value else {
77      let Json::Object(Bind::Lit(object)) = &mut json.value else { return Ok(json) };
78      let mut evaluated = JObject::default();
79      for kv in object.iter_mut() {
80        evaluated.insert(mem::take(&mut kv.0), self.eval(mem::take(&mut kv.1), info)?);
81      }
82      return Ok(JsonWithPos { value: Json::Object(Bind::Lit(evaluated)), pos: json.pos });
83    };
84    self.validate_args("function call", true, 1, list.len(), &json.pos)?;
85    let first_elem = mem::take(list.first_mut().ok_or(ERR)?);
86    let first = self.eval(first_elem.clone(), info)?;
87    if let Json::String(Bind::Lit(cmd)) = &first.value {
88      if self.builtin.contains_key(cmd) {
89        let builtin = self.builtin.get_mut(cmd).ok_or(ERR)?;
90        let scoped = builtin.scoped;
91        let func = builtin.func;
92        let mut tmp_body = vec![];
93        let mut tmp_free_map = BTreeMap::new();
94        let mut tmp_stack_size = 0;
95        if scoped {
96          self.vars.push(HashMap::new());
97          scope_begin(&mut tmp_body, &mut tmp_free_map, &mut tmp_stack_size, info)?;
98        }
99        let rest = list.get_mut(1..).unwrap_or(&mut []).to_vec();
100        let args = if builtin.skip_eval { rest } else { self.eval_args(rest, info)? };
101        let result = func(self, &first, args, info)?;
102        if scoped {
103          scope_end(&mut tmp_body, &mut tmp_free_map, &mut tmp_stack_size, info)?;
104          self.vars.pop();
105        }
106        Ok(JsonWithPos { value: result, pos: first_elem.pos })
107      } else if let Ok(Json::Function(af)) = self.get_var(cmd, &first.pos) {
108        call_func(first_elem.pos, &af, info)
109      } else {
110        err!(self, first.pos, "The `{cmd}` function is undefined.")
111      }
112    } else if let Json::Function(af) = &first.value {
113      call_func(first_elem.pos, af, info)
114    } else {
115      self.typ_err(1, "function call", "LString` or `Function", &first_elem)?;
116      Ok(JsonWithPos::default())
117    }
118  }
119  /// Evaluate arguments.
120  fn eval_args(&mut self, args: Args, info: &mut FuncInfo) -> ErrOR<Vec<JsonWithPos>> {
121    let mut result = vec![];
122    for arg in args {
123      result.push(self.eval(arg, info)?);
124    }
125    Ok(result)
126  }
127  /// Generates a unique name for internal use.
128  pub(crate) fn get_global(&mut self, name: &str, value: &str) -> ErrOR<String> {
129    let seed = self
130      .symbol_seeds
131      .get(name)
132      .map_or(Ok(0), |current| current.checked_add(1).ok_or("Seed Overflow"))?;
133    self.symbol_seeds.insert(name.to_owned(), seed);
134    let l_name = format!(".L{name}{seed:x}");
135    let name_fmt = format!("qword ptr {l_name}[rip]");
136    match name {
137      "BSS" => self.sect.bss.push(format!("  .lcomm {l_name}, {value}\n")),
138      "STR" => {
139        if let Some(str_seed) = self.str_cache.get(value) {
140          return Ok(format!("qword ptr .LSTR{str_seed:x}[rip]"));
141        }
142        self.str_cache.insert(value.to_owned(), self.symbol_seeds.len());
143        self.sect.data.push(format!("  {l_name}: .string \"{value}\"\n"));
144      }
145      "INT" => self.sect.data.push(format!("  {l_name}: .quad {value}\n")),
146      "FNC" => return Ok(l_name),
147      _ => return Err(format!("Internal Error: Unrecognized name: {name}").into()),
148    }
149    Ok(name_fmt)
150  }
151  /// Gets variable.
152  pub(crate) fn get_var(&self, var_name: &str, pos: &Position) -> ErrOR<Json> {
153    for scope in self.vars.iter().rev() {
154      if let Some(val) = scope.get(var_name) {
155        return Ok(val.clone());
156      }
157    }
158    err!(self, pos, "Undefined variables: `{var_name}`")
159  }
160  /// Registers a function in the function table.
161  pub(crate) fn register(&mut self, name: &str, flag: (bool, bool), j_func: JFunc) {
162    self.builtin.insert(name.into(), Builtin { skip_eval: flag.0, scoped: flag.1, func: j_func });
163  }
164  /// Converts `Json` (`StringVar` or `String`) to `StringVar`, otherwise return `Err`
165  pub(crate) fn string2var(
166    &mut self, json: JsonWithPos, ordinal: usize, func_name: &str,
167  ) -> ErrOR<String> {
168    if let Json::String(st) = json.value {
169      match st {
170        Bind::Lit(l_str) => self.get_global("STR", &l_str),
171        Bind::Var(var) => Ok(var),
172        Bind::Local(local) | Bind::Tmp(local) => Ok(fmt_local("qword", local)),
173      }
174    } else {
175      self.typ_err(ordinal, func_name, "String", &json)?;
176      Ok(String::new())
177    }
178  }
179}
180/// Call function and return value.
181fn call_func(pos: Position, af: &AsmFunc, info: &mut FuncInfo) -> ErrOR<JsonWithPos> {
182  info.body.push(format!("  call {}\n", af.name));
183  if let Json::Int(_) = *af.ret {
184    let offset = info.get_local(Align::U64)?;
185    info.body.push(format!("  mov {}, rax\n", fmt_local("qword", offset)));
186    Ok(JsonWithPos { value: Json::Int(Bind::Tmp(offset)), pos })
187  } else {
188    Ok(JsonWithPos { value: Json::Null, pos })
189  }
