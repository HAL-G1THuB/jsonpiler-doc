13/// Generate error.
14macro_rules! err {
15  ($self:ident, $info: expr, $($arg: tt)*) => {
16    Err($self.fmt_err(&format!($($arg)*), &$info).into())
17  };
18}
19impl Jsonpiler {
20  /// Builds the assembly code from the parsed JSON.
21  /// This function is the main entry point for the compilation process.
22  /// It takes the parsed JSON, sets up the initial function table,
23  /// evaluates the JSON, and writes the resulting assembly code to a file.
24  /// # Arguments
25  /// * `source` - The JSON String.
26  /// * `json_file` - The name of the original JSON file.
27  /// * `filename` - The name of the file to write the assembly code to.
28  /// # Returns
29  /// * `Ok(Json)` - The result of the evaluation.
30  /// * `Err(Box<dyn Error>)` - If an error occurred during the compilation process.
31  /// # Errors
32  /// * `Box<dyn Error>` - If an error occurred during the compilation process.
33  #[inline]
34  pub fn build(&mut self, source: String, json_file: &str, filename: &str) -> ErrOR<()> {
35    let json = self.parse(source)?;
36    self.include_flag = HashSet::new();
37    self.sect = Section::default();
38    self.symbol_seeds = HashMap::new();
39    self.builtin = HashMap::new();
40    self.vars = vec![HashMap::new()];
41    self.all_register();
42    let mut func = FuncInfo::default();
43    let result = self.eval(&json, &mut func)?;
44    let mut writer = io::BufWriter::new(File::create(filename)?);
45    writer.write_all(format!(".file \"{json_file}\"\n.intel_syntax noprefix\n").as_bytes())?;
46    writer.write_all(include_bytes!("asm/sect/data.s"))?;
47    for data in &mut self.sect.data {
48      writer.write_all(data.as_bytes())?;
49      writer.write_all(b"\n")?;
50    }
51    writer.write_all(include_bytes!("asm/sect/bss.s"))?;
52    for bss in &mut self.sect.bss {
53      writer.write_all(bss.as_bytes())?;
54      writer.write_all(b"\n")?;
55    }
56    writer.write_all(include_bytes!("asm/sect/start.s"))?;
57    writer.write_all(
58      format!(include_str!("asm/common/prologue.s"), size = func.calc_alloc(8)?).as_bytes(),
59    )?;
60    writer.write_all(b"\n")?;
61    writer.write_all(include_bytes!("asm/sect/startup.s"))?;
62    for body in &mut func.body {
63      writer.write_all(body.as_bytes())?;
64      writer.write_all(b"\n")?;
65    }
66    if let JValue::LInt(int) = result.value {
67      writer.write_all(format!("  mov rcx, {int}\n").as_bytes())
68    } else if let JValue::VInt(var) = &result.value {
69      writer.write_all(format!("  mov rcx, qword ptr {var}[rip]\n").as_bytes())
70    } else {
71      writer.write_all(b"  xor ecx, ecx\n")
72    }?;
73    writer.write_all(b"  call [qword ptr __imp_ExitProcess[rip]]\n.seh_endproc\n")?;
74    writer.write_all(include_bytes!("asm/sect/handler.s"))?;
75    for text in &mut self.sect.text {
76      writer.write_all(text.as_bytes())?;
77      writer.write_all(b"\n")?;
78    }
79    writer.flush()?;
80    Ok(())
81  }
82  /// Evaluates a JSON object.
83  pub(crate) fn eval(&mut self, json: &Json, func: &mut FuncInfo) -> JResult {
84    const ERR: &str = "Unreachable (eval)";
85    let JValue::LArray(list) = &json.value else {
86      let JValue::LObject(object) = &json.value else { return Ok(json.clone()) };
87      let mut evaluated = JObject::default();
88      for kv in object.iter() {
89        evaluated.insert(kv.0.clone(), self.eval(&kv.1, func)?);
90      }
91      return Ok(Json { value: JValue::LObject(evaluated), info: json.info.clone() });
92    };
93    let Some(first_elem) = list.first() else {
94      self.require("A function call", "list with name of function as the first element", json)?;
95      return Ok(Json::default());
96    };
97    let first = &self.eval(first_elem, func)?;
98    if let JValue::LString(cmd) = &first.value {
99      if self.builtin.contains_key(cmd.as_str()) {
100        let builtin = self.builtin.get_mut(cmd.as_str()).ok_or(ERR)?;
101        let scoped = builtin.scoped;
102        let ptr = builtin.func;
103        if scoped {
104          self.vars.push(HashMap::new());
105        }
106        let rest = list.get(1..).unwrap_or(&[]);
107        let args = if builtin.do_not_eval { rest } else { &self.eval_args(rest, func)? };
108        let result = Ok(Json { value: ptr(self, first, args, func)?, info: first.info.clone() });
109        if scoped {
110          self.vars.pop();
111        }
112        result
113      } else {
114        Err(self.fmt_err(&format!("Function `{cmd}` is undefined."), &first.info).into())
115      }
116    } else if let JValue::Function(AsmFunc { name: n, ret: re, .. }) = &first.value {
117      func.body.push(format!("  call {n}"));
118      if let JValue::VInt(_) | JValue::LInt(_) = **re {
119        let na = self.get_name("INT")?;
120        self.sect.bss.push(format!("  .lcomm {na}, 8"));
121        func.body.push(format!("  mov qword ptr {na}[rip], rax"));
122        Ok(Json { value: JValue::VInt(na), info: first.info.clone() })
123      } else {
124        Ok(Json::default())
125      }
126    } else {
127      self.require("1st element of a function call", "function or built-in function", json)?;
128      Ok(Json::default())
129    }
130  }
131  /// Evaluate arguments.
132  fn eval_args(&mut self, args: &[Json], function: &mut FuncInfo) -> ErrOR<Vec<Json>> {
133    let mut result = vec![];
134    for arg in args {
135      result.push(self.eval(arg, function)?);
136    }
137    Ok(result)
138  }
139  /// Generates a unique name for internal use.
140  pub(crate) fn get_name(&mut self, name: &str) -> Result<String, &'static str> {
141    let seed = self
142      .symbol_seeds
143      .get(name)
144      .map_or(Ok(0), |current| current.checked_add(1).ok_or("Seed Overflow"))?;
145    self.symbol_seeds.insert(name.to_owned(), seed);
146    Ok(format!(".L{name}{seed:x}"))
147  }
148  /// Registers a function in the function table.
149  pub(crate) fn register(&mut self, name: &str, flg: (bool, bool), fu: JFunc) {
150    self.builtin.insert(name.into(), BuiltinFunc { do_not_eval: flg.0, scoped: flg.1, func: fu });
151  }
152  /// Generate an error.
153  pub(crate) fn require(&self, name: &str, expected: &str, json: &Json) -> FResult {
154    err!(self, &json.info, "{name} requires {expected}, but got {}.", &json.value)
155  }
156  /// Convert `JValue::` (`StringVar` or `String`) to `StringVar`, otherwise return `Err`
157  pub(crate) fn string2var(&mut self, json: &Json, ctx: &str) -> ErrOR<String> {
158    if let JValue::LString(st) = &json.value {
159      let name = self.get_name("STR")?;
160      self.sect.data.push(format!("  {name}: .string \"{st}\""));
161      Ok(name)
162    } else if let JValue::VString(var) = &json.value {
163      Ok(var.clone())
164    } else {
165      err!(self, &json.info, "`{ctx}` must be a string.")
166    }
167  }
168  /// Generate an error.
169  pub(crate) fn validate(
170    &self, name: &str, at_least: bool, expected: usize, got: usize, info: &ErrorInfo,
171  ) -> ErrOR<()> {
172    let plural = if expected == 1 { "" } else { "s" };
173    if at_least {
174      if got >= expected {
175        Ok(())
176      } else {
177        err!(self, info, "`{name}` requires at least {expected} argument{plural}, but got {got}.")
178      }
179    } else if expected == got {
180      Ok(())
181    } else {
182      err!(self, info, "`{name}` requires exactly {expected} argument{plural}, but got {got}.")
183    }
184  }
