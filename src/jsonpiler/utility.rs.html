1//! Implementation for `Jsonpiler` utility functions
2use crate::{ErrOR, FuncInfo, Json, JsonWithPos, Jsonpiler, Position, err};
3use core::mem::take;
4impl Jsonpiler {
5  /// Format error.
6  #[must_use]
7  pub(crate) fn fmt_err(&self, err: &str, pos: &Position) -> String {
8    let gen_err = |msg: &str| -> String {
9      format!("{err}\nError occurred on line: {}\nError position:\n{msg}", pos.line)
10    };
11    if self.source.is_empty() {
12      return gen_err("\n^");
13    }
14    let len = self.source.len();
15    let idx = pos.offset.min(len.saturating_sub(1));
16    let start = if idx == 0 {
17      0
18    } else {
19      let Some(left) = self.source.get(..idx) else {
20        return gen_err("Error: Failed to get substring");
21      };
22      match left.rfind('\n') {
23        None => 0,
24        Some(start_offset) => {
25          let Some(res) = start_offset.checked_add(1) else {
26            return gen_err("Error: Overflow");
27          };
28          res
29        }
30      }
31    };
32    let Some(right) = self.source.get(idx..) else {
33      return gen_err("Error: Failed to get substring");
34    };
35    let end = match right.find('\n') {
36      None => len,
37      Some(end_offset) => {
38        let Some(res) = idx.checked_add(end_offset) else {
39          return gen_err("Error: Overflow");
40        };
41        res
42      }
43    };
44    let ws = " ".repeat(idx.saturating_sub(start));
45    let Some(result) = self.source.get(start..end) else {
46      return gen_err("Error: Failed to get substring");
47    };
48    gen_err(&format!("{result}\n{ws}^"))
49  }
50  /// Generates a type error.
51  pub(crate) fn typ_err(
52    &self, ordinal: usize, name: &str, expected: &str, json: &JsonWithPos,
53  ) -> ErrOR<Json> {
54    let suffix = match ordinal % 100 {
55      11..=13 => "th",
56      _ => match ordinal % 10 {
57        1 => "st",
58        2 => "nd",
59        3 => "rd",
60        _ => "th",
61      },
62    };
63    let typ = json.value.type_name();
64    err!(
65      self,
66      &json.pos,
67      "The {ordinal}{suffix} argument to `{name}` must be of a type `{expected}`, \
68      but a value of type `{typ}` was provided."
69    )
70  }
71  /// Generate an error.
72  pub(crate) fn validate_args(
73    &self, name: &str, at_least: bool, expected: usize, supplied: usize, pos: &Position,
74  ) -> ErrOR<()> {
75    let fmt_require = |text: &str| -> ErrOR<()> {
76      let (plural, be) = if expected == 1 { ("", "is") } else { ("s", "are") };
77      err!(
78        self,
79        pos,
80        "`{name}` requires {text} {expected} argument{plural}, \
81        but {supplied} argument{plural} {be} supplied.",
82      )
83    };
84    if at_least {
85      if supplied >= expected { Ok(()) } else { fmt_require("at least") }
86    } else if expected == supplied {
87      Ok(())
88    } else {
89      fmt_require("exactly")
90    }
91  }
92}
93/// Format local variables.
94pub(crate) fn fmt_local(size: &str, offset: usize) -> String {
95  format!("{size} ptr -{offset}[rbp]")
96}
97/// Begin scope.
98#[expect(clippy::single_call_fn, reason = "")]
99pub(crate) fn scope_begin(tmp: &mut FuncInfo, info: &mut FuncInfo) -> ErrOR<()> {
100  info.scope_align = info
101    .scope_align
102    .checked_add(info.stack_size.checked_add(15).ok_or("Layout Overflow")? & !15)
103    .ok_or("ScopeAlign Overflow")?;
104  tmp.body = take(&mut info.body);
105  tmp.free_map = take(&mut info.free_map);
106  tmp.stack_size = take(&mut info.stack_size);
107  Ok(())
108}
109/// Begin scope.
110#[expect(clippy::single_call_fn, reason = "")]
111pub(crate) fn scope_end(tmp: &mut FuncInfo, info: &mut FuncInfo) -> ErrOR<()> {
112  let align = info.stack_size.checked_add(15).ok_or("Overflow local size")? & !15;
113  if align != 0 {
114    tmp.body.push(format!("  sub rsp, {align}\n"));
115  }
116  tmp.body.append(&mut info.body);
117  if align != 0 {
118    tmp.body.push(format!("  add rsp, {align}\n"));
119  }
120  info.body = take(&mut tmp.body);
121  info.free_map = take(&mut tmp.free_map);
122  info.stack_size = take(&mut tmp.stack_size);
123  Ok(())
124}
