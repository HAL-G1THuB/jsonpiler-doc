1//! Implementation of the `JValue`
2use super::{JValue, Json, functions::escape_string};
3use core::fmt;
4
5impl fmt::Display for JValue {
6  /// Formats the `Json` object as a compact string without indentation.
7  #[expect(clippy::min_ident_chars, reason = "default name is 'f'")]
8  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
9    match self {
10      JValue::Null => f.write_str("null"),
11      JValue::LBool(bo) => write!(f, "{bo}"),
12      JValue::VBool(bv, bit) => write!(f, "({bv}-{bit}: bool)"),
13      JValue::LInt(int) => write!(f, "{int}"),
14      JValue::VInt(iv) => write!(f, "({iv}: int)"),
15      JValue::LFloat(fl) => write!(f, "{fl}"),
16      JValue::VFloat(fv) => write!(f, "({fv}: float)"),
17      JValue::LString(st) => f.write_str(&escape_string(st)?),
18      JValue::VString(sv) => write!(f, "({sv}: string)"),
19      JValue::LArray(ar) => {
20        f.write_str("[")?;
21        iter_write(ar, f)?;
22        f.write_str("]")
23      }
24      JValue::VArray(av) => write!(f, "({av}: array)"),
25      JValue::Function { name: na, params: pa, ret: re } => {
26        write!(f, "{na}(")?;
27        iter_write(pa, f)?;
28        write!(f, ") -> ")?;
29        (*re).clone().fmt(f)
30      }
31      JValue::LObject(obj) => {
32        f.write_str("{")?;
33        for (i, kv) in obj.iter().enumerate() {
34          if i > 0 {
35            f.write_str(", ")?;
36          }
37          write!(f, "{}: ", escape_string(&kv.0)?)?;
38          kv.1.value.fmt(f)?;
39        }
40        f.write_str("}")
41      }
42      JValue::VObject(ov) => write!(f, "({ov}: object)"),
43    }
44  }
45}
46
47/// Iterates over a list of `Json` objects and writes them without indentation.
48fn iter_write(list: &[Json], out: &mut fmt::Formatter) -> fmt::Result {
49  for (i, item) in list.iter().enumerate() {
50    if i > 0 {
51      out.write_str(", ")?;
52    }
53    write!(out, "{}", item.value)?;
54  }
55  Ok(())
56}
