1//! Implementation of the `Json`
2use super::{Bind, Json, JsonWithPos};

4impl Json {
5  /// Determines if it is a literal.
6  pub fn is_literal(&self) -> bool {
7    match self {
8      Json::LBool(_) | Json::Null => true,
9      Json::VBool(..) | Json::Function(_) => false,
10      Json::Object(object) => matches!(object, Bind::Lit(_)),
11      Json::Float(float) => matches!(float, Bind::Lit(_)),
12      Json::Int(int) => matches!(int, Bind::Lit(_)),
13      Json::String(string) => matches!(string, Bind::Lit(_)),
14      Json::Array(array) => matches!(array, Bind::Lit(_)),
15    }
16  }
17  /// Determines if it is a temporary value.
18  pub fn tmp(&self) -> Option<(usize, usize)> {
19    match self {
20      Json::LBool(_) | Json::Null | Json::VBool(_) | Json::Function(_) => None,
21      Json::Object(object) => match object {
22        Bind::Tmp(local) => Some((*local, 8)),
23        _ => None,
24      },
25      Json::Float(float) => match float {
26        Bind::Tmp(local) => Some((*local, 8)),
27        _ => None,
28      },
29      Json::Int(int) => match int {
30        Bind::Tmp(local) => Some((*local, 8)),
31        _ => None,
32      },
33      Json::String(string) => match string {
34        Bind::Tmp(local) => Some((*local, 8)),
35        _ => None,
36      },
37      Json::Array(array) => match array {
38        Bind::Tmp(local) => Some((*local, 8)),
39        _ => None,
40      },
41    }
42  }
43  /// Generate type name.
44  pub fn type_name(&self) -> &'static str {
45    match self {
46      Json::LBool(_) => "LBool",
47      Json::Null => "Null",
48      Json::VBool(_) => "VBool",
49      Json::Function(_) => "Function",
50      Json::Object(object) => {
51        if matches!(object, Bind::Lit(_)) {
52          "LObject"
53        } else {
54          "VObject"
55        }
56      }
57      Json::Float(float) => {
58        if matches!(float, Bind::Lit(_)) {
59          "LFloat"
60        } else {
61          "VFloat"
62        }
63      }
64      Json::Int(int) => {
65        if matches!(int, Bind::Lit(_)) {
66          "LInt"
67        } else {
68          "VInt"
69        }
70      }
71      Json::String(string) => {
72        if matches!(string, Bind::Lit(_)) {
73          "LString"
74        } else {
75          "VString"
76        }
77      }
78      Json::Array(array) => {
79        if matches!(array, Bind::Lit(_)) {
80          "LArray"
81        } else {
82          "VArray"
83        }
84      }
85    }
86  }
87}
88impl fmt::Display for Json {
89  /// Formats the `Json` object as a compact string without indentation.
90  #[expect(clippy::min_ident_chars, reason = "default name is 'f'")]
91  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
92    match self {
93      Json::Null => f.write_str("Null"),
94      Json::Array(ar) => match ar {
95        Bind::Lit(array) => {
96          f.write_str("[")?;
97          iter_write(array, f)?;
98          f.write_str("]")
99        }
100        Bind::Var(var) => write!(f, "VArray(\"{var}\")"),
101        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VArray(\"{local}\")"),
102      },
103      Json::LBool(bo) => write!(f, "{bo}"),
104      Json::VBool(vb) => write!(f, "VBool(\"{}\"-{})", vb.name, vb.bit),
105      Json::Int(int) => match int {
106        Bind::Lit(l_int) => write!(f, "{l_int}"),
107        Bind::Var(var) => write!(f, "VInt(\"{var}\")"),
108        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VInt(\"{local}\")"),
109      },
110      Json::Float(float) => match float {
111        Bind::Lit(flt) => write!(f, "{flt}"),
112        Bind::Var(var) => write!(f, "VFloat(\"{var}\")"),
113        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VFloat(\"{local}\")"),
114      },
115      Json::String(st) => match st {
116        Bind::Lit(lstr) => f.write_str(&escape_string(lstr)?),
117        Bind::Var(var) => write!(f, "VString(\"{var}\")"),
118        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VString(\"{local}\")"),
119      },
120      Json::Function(fu) => {
121        write!(f, "{}(", fu.name)?;
122        iter_write(&fu.params, f)?;
123        write!(f, ") -> ")?;
124        (*fu.ret).clone().fmt(f)
125      }
126      Json::Object(object) => match object {
127        Bind::Lit(obj) => {
128          f.write_str("{")?;
129          for (i, kv) in obj.iter().enumerate() {
130            if i > 0 {
131              f.write_str(", ")?;
132            }
133            write!(f, "{}: ", escape_string(&kv.0)?)?;
134            kv.1.value.fmt(f)?;
135          }
136          f.write_str("}")
137        }
138        Bind::Var(var) => write!(f, "VObject({var})"),
139        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VObject(\"{local}\")"),
140      },
141    }
142  }
143}
144/// Escapes special characters in a string for proper JSON formatting.
145pub(crate) fn escape_string(unescaped: &str) -> Result<String, fmt::Error> {
146  let mut escaped = String::new();
147  escaped.push('"');
148  for ch in unescaped.chars() {
149    match ch {
150      '"' => write!(escaped, r#"\""#)?,
151      '\\' => write!(escaped, r"\\")?,
152      '\n' => write!(escaped, r"\n")?,
153      '\t' => write!(escaped, r"\t")?,
154      '\r' => write!(escaped, r"\r")?,
155      '\u{08}' => write!(escaped, r"\b")?,
156      '\u{0C}' => write!(escaped, r"\f")?,
157      u_ch if u_ch < '\u{20}' => write!(escaped, r"\u{:04x}", u32::from(ch))?,
158      _ => escaped.push(ch),
159    }
160  }
161  escaped.push('"');
162  Ok(escaped)
163}
164/// Iterates over a list of `Json` objects and writes them without indentation.
165fn iter_write(list: &[JsonWithPos], out: &mut fmt::Formatter) -> fmt::Result {
166  for (i, item) in list.iter().enumerate() {
167    if i > 0 {
168      out.write_str(", ")?;
169    }
170    write!(out, "{}", item.value)?;
171  }
172  Ok(())
173}
