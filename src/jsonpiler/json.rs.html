7impl Json {
8  /// Determines if it is a literal.
9  pub fn is_literal(&self) -> bool {
10    match self {
11      Json::LBool(_) | Json::Null => true,
12      Json::VBool(..) | Json::Function(_) => false,
13      Json::Object(object) => matches!(object, Bind::Lit(_)),
14      Json::Float(float) => matches!(float, Bind::Lit(_)),
15      Json::Int(int) => matches!(int, Bind::Lit(_)),
16      Json::String(string) => matches!(string, Bind::Lit(_)),
17      Json::Array(array) => matches!(array, Bind::Lit(_)),
18    }
19  }
20  /// Determines if it is a temporary value.
21  pub fn tmp(&self) -> Option<(usize, usize)> {
22    match self {
23      Json::LBool(_) | Json::Null | Json::VBool(_) | Json::Function(_) => None,
24      Json::Object(object) => match object {
25        Bind::Tmp(local) => Some((*local, 8)),
26        _ => None,
27      },
28      Json::Float(float) => match float {
29        Bind::Tmp(local) => Some((*local, 8)),
30        _ => None,
31      },
32      Json::Int(int) => match int {
33        Bind::Tmp(local) => Some((*local, 8)),
34        _ => None,
35      },
36      Json::String(string) => match string {
37        Bind::Tmp(local) => Some((*local, 8)),
38        _ => None,
39      },
40      Json::Array(array) => match array {
41        Bind::Tmp(local) => Some((*local, 8)),
42        _ => None,
43      },
44    }
45  }
46  /// Converts temporary value to Local variable.
47  pub fn tmp_to_local(&mut self) -> Self {
48    match self {
49      Json::Object(Bind::Tmp(local)) => Json::Object(Bind::Local(*local)),
50      Json::Array(Bind::Tmp(local)) => Json::Array(Bind::Local(*local)),
51      Json::Float(Bind::Tmp(local)) => Json::Float(Bind::Local(*local)),
52      Json::String(Bind::Tmp(local)) => Json::String(Bind::Local(*local)),
53      Json::Int(Bind::Tmp(local)) => Json::Int(Bind::Local(*local)),
54      _ => take(self),
55    }
56  }
57  /// Generate type name.
58  pub fn type_name(&self) -> String {
59    match self {
60      Json::LBool(_) => "LBool".to_owned(),
61      Json::Null => "Null".to_owned(),
62      Json::VBool(_) => "VBool".to_owned(),
63      Json::Function(_) => "Function".to_owned(),
64      Json::Float(float) => bind_name(float, "Float"),
65      Json::Object(obj) => bind_name(obj, "Object"),
66      Json::Int(int) => bind_name(int, "Int"),
67      Json::String(st) => bind_name(st, "String"),
68      Json::Array(arr) => bind_name(arr, "Array"),
69    }
70  }
71}
72impl fmt::Display for Json {
73  /// Formats the `Json` object as a compact string without indentation.
74  #[expect(clippy::min_ident_chars, reason = "default name is 'f'")]
75  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
76    match self {
77      Json::Null => f.write_str("Null"),
78      Json::Array(ar) => match ar {
79        Bind::Lit(array) => {
80          f.write_str("[")?;
81          iter_write(array, f)?;
82          f.write_str("]")
83        }
84        Bind::Var(var) => write!(f, "VArray(\"{var}\")"),
85        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VArray(\"{local}\")"),
86      },
87      Json::LBool(bo) => write!(f, "{bo}"),
88      Json::VBool(vb) => write!(f, "VBool(\"{}\"-{})", vb.name, vb.bit),
89      Json::Int(int) => match int {
90        Bind::Lit(l_int) => write!(f, "{l_int}"),
91        Bind::Var(var) => write!(f, "VInt(\"{var}\")"),
92        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VInt(\"{local}\")"),
93      },
94      Json::Float(float) => match float {
95        Bind::Lit(flt) => write!(f, "{flt}"),
96        Bind::Var(var) => write!(f, "VFloat(\"{var}\")"),
97        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VFloat(\"{local}\")"),
98      },
99      Json::String(st) => match st {
100        Bind::Lit(lstr) => f.write_str(&escape_string(lstr)?),
101        Bind::Var(var) => write!(f, "VString(\"{var}\")"),
102        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VString(\"{local}\")"),
103      },
104      Json::Function(fu) => {
105        write!(f, "{}(", fu.name)?;
106        iter_write(&fu.params, f)?;
107        write!(f, ") -> ")?;
108        (*fu.ret).clone().fmt(f)
109      }
110      Json::Object(object) => match object {
111        Bind::Lit(obj) => {
112          f.write_str("{")?;
113          for (i, kv) in obj.iter().enumerate() {
114            if i > 0 {
115              f.write_str(", ")?;
116            }
117            write!(f, "{}: ", escape_string(&kv.0)?)?;
118            kv.1.value.fmt(f)?;
119          }
120          f.write_str("}")
121        }
122        Bind::Var(var) => write!(f, "VObject({var})"),
123        Bind::Local(local) | Bind::Tmp(local) => write!(f, "VObject(\"{local}\")"),
124      },
125    }
126  }
127}
128/// gets name of the `Bind`.
129fn bind_name<T>(bind: &Bind<T>, ty: &str) -> String {
130  let mut l_or_v = if matches!(bind, Bind::Lit(_)) { "L" } else { "V" }.to_owned();
131  l_or_v.push_str(ty);
132  l_or_v
133}
134/// Escapes special characters in a string for proper JSON formatting.
135pub(crate) fn escape_string(unescaped: &str) -> Result<String, fmt::Error> {
136  let mut escaped = String::new();
137  escaped.push('"');
138  for ch in unescaped.chars() {
139    match ch {
140      '"' => write!(escaped, r#"\""#)?,
141      '\\' => write!(escaped, r"\\")?,
142      '\n' => write!(escaped, r"\n")?,
143      '\t' => write!(escaped, r"\t")?,
144      '\r' => write!(escaped, r"\r")?,
145      '\u{08}' => write!(escaped, r"\b")?,
146      '\u{0C}' => write!(escaped, r"\f")?,
147      u_ch if u_ch < '\u{20}' => write!(escaped, r"\u{:04x}", u32::from(ch))?,
148      _ => escaped.push(ch),
149    }
150  }
151  escaped.push('"');
152  Ok(escaped)
153}
154/// Iterates over a list of `Json` objects and writes them without indentation.
155fn iter_write(list: &[JsonWithPos], out: &mut fmt::Formatter) -> fmt::Result {
156  for (i, item) in list.iter().enumerate() {
157    if i > 0 {
158      out.write_str(", ")?;
159    }
160    write!(out, "{}", item.value)?;
161  }
162  Ok(())
