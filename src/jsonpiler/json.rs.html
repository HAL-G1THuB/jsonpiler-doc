1//! Implementation of the `Json`
2use super::{

6use core::{

10impl Json {
11  /// Determines if it is a temporary value.
12  pub fn tmp(&self) -> Option<(usize, usize)> {
13    match self {
14      Json::LBool(_) | Json::Null | Json::VBool(_) | Json::Function(_) => None,
15      Json::Object(bind) => bind_match(bind, 8),
16      Json::Float(bind) => bind_match(bind, 8),
17      Json::Int(bind) => bind_match(bind, 8),
18      Json::String(bind) => bind_match(bind, 8),
19      Json::Array(bind) => bind_match(bind, 8),
20    }
21  }
22  /// Converts temporary value to Local variable.
23  pub fn tmp_to_local(&mut self) -> Self {
24    match *self {
25      Json::Object(Bind::Var(Name { var: Tmp, seed })) => {
26        Json::Object(Bind::Var(Name { var: Local, seed }))
27      }
28      Json::Array(Bind::Var(Name { var: Tmp, seed })) => {
29        Json::Array(Bind::Var(Name { var: Local, seed }))
30      }
31      Json::Float(Bind::Var(Name { var: Tmp, seed })) => {
32        Json::Float(Bind::Var(Name { var: Local, seed }))
33      }
34      Json::String(Bind::Var(Name { var: Tmp, seed })) => {
35        Json::String(Bind::Var(Name { var: Local, seed }))
36      }
37      Json::Int(Bind::Var(Name { var: Tmp, seed })) => {
38        Json::Int(Bind::Var(Name { var: Local, seed }))
39      }
40      Json::VBool(AsmBool { name: Name { var: Tmp, seed }, bit }) => {
41        Json::VBool(AsmBool { name: Name { var: Local, seed }, bit })
42      }
43      Json::String(_)
44      | Json::Int(_)
45      | Json::Object(_)
46      | Json::Float(_)
47      | Json::Array(_)
48      | Json::LBool(_)
49      | Json::VBool(_)
50      | Json::Function(_)
51      | Json::Null => take(self),
52    }
53  }
54  /// Generate type name.
55  pub fn type_name(&self) -> String {
56    match self {
57      Json::LBool(_) => "LBool".to_owned(),
58      Json::Null => "Null".to_owned(),
59      Json::VBool(_) => "VBool".to_owned(),
60      Json::Function(_) => "Function".to_owned(),
61      Json::Float(float) => bind_name(float, "Float"),
62      Json::Object(obj) => bind_name(obj, "Object"),
63      Json::Int(int) => bind_name(int, "Int"),
64      Json::String(st) => bind_name(st, "String"),
65      Json::Array(arr) => bind_name(arr, "Array"),
66    }
67  }
68  /// Determines if it is a global variable.
69  pub fn var(&self) -> Option<VarKind> {
70    match self {
71      Json::Function(_) => Some(Global),
72      Json::VBool(AsmBool { name, .. })
73      | Json::Object(Bind::Var(name))
74      | Json::Float(Bind::Var(name))
75      | Json::Int(Bind::Var(name))
76      | Json::String(Bind::Var(name))
77      | Json::Array(Bind::Var(name)) => Some(name.var),
78      Json::String(_)
79      | Json::Int(_)
80      | Json::Object(_)
81      | Json::Float(_)
82      | Json::Array(_)
83      | Json::LBool(_)
84      | Json::Null => None,
85    }
86  }
87}
88impl fmt::Display for Json {
89  /// Formats the `Json` object as a compact string without indentation.
90  #[expect(clippy::min_ident_chars, reason = "default name is 'f'")]
91  fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
92    match self {
93      Json::Null => f.write_str("Null"),
94      Json::Array(ar) => match ar {
95        Bind::Lit(array) => {
96          f.write_str("[")?;
97          iter_write(array, f)?;
98          f.write_str("]")
99        }
100        Bind::Var(name) => write!(f, "VArray(qword{name})"),
101      },
102      Json::LBool(bo) => write!(f, "{bo}"),
103      Json::VBool(asm_bool) => write!(f, "VBool(qword{}-{})", asm_bool.name, asm_bool.bit),
104      Json::Int(int) => match int {
105        Bind::Lit(l_int) => write!(f, "{l_int}"),
106        Bind::Var(name) => write!(f, "VInt(qword{name})"),
107      },
108      Json::Float(float) => match float {
109        Bind::Lit(l_float) => write!(f, "{l_float}"),
110        Bind::Var(name) => write!(f, "VFloat(qword{name})"),
111      },
112      Json::String(st) => match st {
113        Bind::Lit(l_st) => f.write_str(&escape_string(l_st)?),
114        Bind::Var(name) => write!(f, "VString(qword{name})"),
115      },
116      Json::Function(fu) => {
117        write!(f, "{}(", fu.name)?;
118        iter_write(&fu.params, f)?;
119        write!(f, ") -> ")?;
120        (*fu.ret).clone().fmt(f)
121      }
122      Json::Object(object) => match object {
123        Bind::Lit(obj) => {
124          f.write_str("{")?;
125          for (i, kv) in obj.iter().enumerate() {
126            if i > 0 {
127              f.write_str(", ")?;
128            }
129            write!(f, "{}: ", escape_string(&kv.0)?)?;
130            kv.1.value.fmt(f)?;
131          }
132          f.write_str("}")
133        }
134        Bind::Var(name) => write!(f, "VObject(qword{name})"),
135      },
136    }
137  }
138}
139/// Pattern match of the `Bind`.
140fn bind_match<T>(bind: &Bind<T>, byte: usize) -> Option<(usize, usize)> {
141  match bind {
142    Bind::Var(name) => Some((name.seed, byte)),
143    Bind::Lit(_) => None,
144  }
145}
146/// gets name of the `Bind`.
147fn bind_name<T>(bind: &Bind<T>, ty: &str) -> String {
148  let mut l_or_v = if matches!(bind, Bind::Lit(_)) { "L" } else { "V" }.to_owned();
149  l_or_v.push_str(ty);
150  l_or_v
151}
152/// Escapes special characters in a string for proper JSON formatting.
153pub(crate) fn escape_string(unescaped: &str) -> Result<String, fmt::Error> {
154  let mut escaped = String::new();
155  escaped.push('"');
156  for ch in unescaped.chars() {
157    match ch {
158      '"' => write!(escaped, r#"\""#)?,
159      '\\' => write!(escaped, r"\\")?,
160      '\n' => write!(escaped, r"\n")?,
161      '\t' => write!(escaped, r"\t")?,
162      '\r' => write!(escaped, r"\r")?,
163      '\u{08}' => write!(escaped, r"\b")?,
164      '\u{0C}' => write!(escaped, r"\f")?,
165      u_ch if u_ch < '\u{20}' => write!(escaped, r"\u{:04x}", u32::from(ch))?,
166      _ => escaped.push(ch),
167    }
168  }
169  escaped.push('"');
170  Ok(escaped)
171}
172/// Iterates over a list of `Json` objects and writes them without indentation.
173fn iter_write(list: &[JsonWithPos], out: &mut fmt::Formatter) -> fmt::Result {
174  for (i, item) in list.iter().enumerate() {
175    if i > 0 {
176      out.write_str(", ")?;
177    }
178    write!(out, "{}", item.value)?;
179  }
180  Ok(())
