1//! Implementation for `FuncInfo`.
2use {super::FuncInfo, core::cmp};
3impl FuncInfo {
4  /// Add local variable size (in bytes)
5  #[expect(dead_code, reason = "todo")]
6  pub fn add_local(&mut self, size: usize) -> Result<(), &'static str> {
7    self.local_size = self.local_size.checked_add(size).ok_or("LocalSize Overflow")?;
8    Ok(())
9  }
10  /// Calculate to allocate size.
11  pub fn calc_alloc(&self, align: usize) -> Result<usize, &'static str> {
12    let args_size = self.args_slots.checked_mul(8).ok_or("Overflow: args_slots * 8")?;
13    let raw = self.local_size.checked_add(args_size).ok_or("Overflow: local_size + args")?;
14    let locals = raw.checked_add(15).ok_or("Overflow before alignment")? & !15;
15    let shadow_space = 32usize.checked_add(align).ok_or("ShadowSpace Overflow")?;
16    locals.checked_add(shadow_space).ok_or("LocalSize Overflow")
17  }
18  /// Update `args_slots` (only if size is larger)
19  #[expect(dead_code, reason = "todo")]
20  pub fn update_max(&mut self, size: usize) {
21    self.args_slots = cmp::max(self.args_slots, size);
22  }
23}
