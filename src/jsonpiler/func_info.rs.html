1//! Implementation for `FuncInfo`.
2use crate::{ErrOR, FuncInfo, Name, Var::Local, add};
3use core::cmp;
4impl FuncInfo {
5  /// Calculate to allocate size.
6  pub fn calc_alloc(&self, align: usize) -> ErrOR<usize> {
7    let args_size = self.args_slots.checked_mul(8).ok_or("Overflow: args_slots * 8")?;
8    let raw = add(self.stack_size, args_size)?;
9    let locals = add(raw, 15)? & !15;
10    let shadow_space = add(align, 32)?;
11    add(locals, shadow_space)
12  }
13  /// Free from stack.
14  pub fn free(&mut self, end: usize, mut size: usize) -> ErrOR<()> {
15    let mut start = end.checked_sub(size).ok_or("StackCalc Overflow")?;
16    if let Some((&prev_start, &prev_size)) = self.free_map.range(..start).next_back() {
17      if add(prev_start, prev_size)? == start {
18        self.free_map.remove(&prev_start);
19        start = prev_start;
20        size = add(size, prev_size)?;
21      }
22    }
23    if let Some((&next_start, &next_size)) = self.free_map.range(start..).next() {
24      if end == next_start {
25        self.free_map.remove(&next_start);
26        size = add(size, next_size)?;
27      }
28    }
29    self.free_map.insert(start, size);
30    Ok(())
31  }
32  /// get local variable name.
33  pub fn get_local(&mut self, byte: usize) -> ErrOR<Name> {
34    Ok(Name { var: Local, seed: add(self.push(byte)?, self.scope_align)? })
35  }
36  /// Push to stack.
37  fn push(&mut self, byte: usize) -> ErrOR<usize> {
38    let sub = |op1: usize, op2: usize| -> Result<usize, &'static str> {
39      op1.checked_sub(op2).ok_or("InternalError: Error in FuncInfo::push")
40    };
41    let dec_align = sub(byte, 1)?;
42    for (&start, &size) in &self.free_map {
43      let aligned_start = add(start, dec_align)? & !dec_align;
44      let padding = sub(aligned_start, start)?;
45      if size >= add(padding, byte)? {
46        self.free_map.remove(&start);
47        if padding > 0 {
48          self.free_map.insert(start, padding);
49        }
50        let used_end = add(aligned_start, byte)?;
51        let tail_size = sub(add(start, size)?, used_end)?;
52        if tail_size > 0 {
53          self.free_map.insert(used_end, tail_size);
54        }
55        return Ok(used_end);
56      }
57    }
58    let aligned_start = add(self.stack_size, dec_align)? & !dec_align;
59    if aligned_start > self.stack_size {
60      let gap_size = sub(aligned_start, self.stack_size)?;
61      self.free_map.insert(self.stack_size, gap_size);
62    }
63    let new_end = add(aligned_start, byte)?;
64    self.stack_size = new_end;
65    Ok(new_end)
66  }
67  /// Update `args_slots` (only if size is larger)
68  #[expect(dead_code, reason = "todo")]
69  pub fn update_max(&mut self, size: usize) {
70    self.args_slots = cmp::max(self.args_slots, size);
71  }
72}
